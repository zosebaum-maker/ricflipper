// Ric Flipper — Cloudflare Worker v7 const CORS = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET, POST, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type', }; export default { async fetch(request) { if (request.method === 'OPTIONS') return new Response(null, { headers: CORS }); const url = new URL(request.url); const action = url.searchParams.get('action'); let body = {}; try { body = await request.json(); } catch(e) {} if (action === 'token') return await handleToken(body); if (action === 'search') return await handleSearch(body); if (action === 'pc') return await handlePC(body); return json({ error: 'Unknown action' }, 400); } }; async function handleToken({ appId, certId }) { try { if (!appId || !certId) return json({ error: 'Missing credentials' }, 400); return json(await res.json()); } catch(e) { return json({ error: e.message }, 500); } } async function handleSearch({ token, q, filter, sort, limit, minVal }) { try { const params = new URLSearchParams({ q: q || 'pokemon card graded', category_ids: '183454', filter: `${filter},price:[${minVal||250}..],priceCurrency:USD,conditionIds:{3000}`, sort: sort || 'endingSoonest', limit: String(limit || 100), fieldgroups: 'EXTENDED', }); const res = await fetch( `https://api.ebay.com/buy/browse/v1/item_summary/search?${params}`, { headers: { 'Authorization': `Bearer ${token}`, 'X-EBAY-C-MARKETPLACE-ID': 'EBAY_US' } } ); return json(await res.json()); } catch(e) { return json({ error: e.message }, 500); } } async function handlePC({ pcKey, q, id }) { try { if (!pcKey) return json({ error: 'Missing pcKey' }, 400); // Use ID lookup if available — exact match, most accurate let url; } else { return json({ error: 'Missing q or id' }, 400); } const res = await fetch(url, { headers: { 'User-Agent': 'RicFlipper/1.0' } }); if (!res.ok) return json({ error: 'PC API returned ' + res.status }, 500); const text = await res.text(); try { return json(JSON.parse(text)); } catch(e) { return json({ error: 'PC non-JSON response', raw: text.substring(0, 200) }, 500); } } catch(e) { return json({ error: e.message }, 500); } } function json(data, status = 200) { return new Response(JSON.stringify(data), { status, headers: { ...CORS, 'Content-Type': 'application/json' }, }); }
